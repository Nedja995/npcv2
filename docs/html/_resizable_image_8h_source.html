162  void Resize( const sf::Vector2u& customSize = sf::Vector2u( 0, 0 ) )
163  {
164  // If no size is given (0,0) , use original image...
165  if( !customSize.x || !customSize.y )
166  {
167  Image::SetImage( m_OriginalImage );
168  return;
169  }
170 
171  // Let the flood begin
172  float originalX = static_cast<float>( m_OriginalImage.getSize().x );
173  float originalY = static_cast<float>( m_OriginalImage.getSize().y );
174  float desiredX = static_cast<float>( customSize.x );
175  float desiredY = static_cast<float>( customSize.y );
176 
177  // Compute the scale we will apply to image
178  float scaleX = desiredX / originalX;
179  float scaleY = desiredY / originalY;
180 
181  // Get texture of the original image
182  sf::Texture tempTexture;
183  tempTexture.loadFromImage( m_OriginalImage );
184 
185  // Sprite allows transformations to be made to it, so let's assign our texture to it
186  // (warn: sprite doesn't own the texture, keep it alive till we're done)
187  sf::Sprite tempSprite;
188  tempSprite.setTexture( tempTexture, true );
189 
190 
191  if( m_KeepAspect )
192  {
193  // Use same scale for both sides of the sprite
194  float lowerScale = std::min( scaleX, scaleY );
195 
196  tempSprite.scale( lowerScale, lowerScale );
197 
198 
199  // Since the image will not be stretched out because we want to keep the aspect of the
200  // image, there will be an empty space to the right/down of the resized image. I think
201  // it's better if the empty place is distributed around the resized image on all sides,
202  // so we move the sprite to the center of the rectangle we set by size we want.
203  //
204  // Example
205  // ### ###00
206  // ### ###00
207  // ### ###00
208  // ### ###00
209  // image 5x3 ### resize it to 5x5, keep aspect ###00 0=empty space only on one side
210  //
211  //
212  // 0###0
213  // 0###0
214  // 0###0
215  // 0###0
216  // so center the image 0###0 and now empty space is distributed to both sides
217 
218  // Move the sprite to the center of the resized image
219  float offsetX = ( desiredX - ( originalX * ( lowerScale ) ) ) / 2;
220  float offsetY = ( desiredY - ( originalY * ( lowerScale ) ) ) / 2;
221 
222  tempSprite.move( offsetX, offsetY );
223  }
224  else
225  {
226  tempSprite.scale( scaleX, scaleY );
227  }
228 
229  // Transformations on Sprite are set, so we can pre-render the sprite on
230  // a new texture with a transparent background
231  sf::RenderTexture tempRenderTexture;
232  tempRenderTexture.create( customSize.x, customSize.y );
233  tempRenderTexture.setSmooth( true );
234  tempRenderTexture.clear( sf::Color( 255,255,255,0 ) );
235  tempRenderTexture.draw( tempSprite );
236  tempRenderTexture.display();
237 
238  // Now when we have the new texture with our resized image ready
239  // we can set it as a new image to the underlying base Image
240  // class, and we are done.
241  Image::SetImage( tempRenderTexture.getTexture().copyToImage() );
242  }
